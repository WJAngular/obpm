//跨表单主子流程，主流程文档某控件值为其子实例某字段值的总和(值脚本)
(function(){
var subfdocs = getSubFlowDocList();   //获取子流程文档列表
var rtn = 0;
if (subfdocs!=null){
   for (var it = subfdocs.iterator();it.hasNext();){  //遍历文档
      var doc = it.next();
      rtn += doc.getItemValueAsDouble("子金额");  //字段值累加
}
}
return rtn;
})()
