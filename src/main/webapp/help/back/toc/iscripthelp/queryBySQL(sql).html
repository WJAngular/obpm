返回

queryBySQL(sql)

概述

根据sql查询符合条件的Document，结果以Collection对象返回。

返回值

java.util.Collection

参数

1、sqlString

查询符合条件语句

示例

描述:

实现queryBySQL(sql)函数的应用，查询文档记录(SQL)



iScript 代码:


//表单某字段值为根据SQL姓名字段的值查询出对应的电话(值脚本)
(function(){
var user = getItemValue("SQL姓名");
var rtn = '';
if(user!=null && user.trim().length()>0){
var sql="select * from tlk_查询文档记录SQL where item_SQL姓名='"+user+"'";
var query = queryBySQL(sql);
if(query!=null){
	      var iter  =query.iterator();
	      if(iter != null && iter.hasNext()){
		       doc = iter.next();
		       rtn = doc.getItemValueAsString("SQL电话");   
	      }	
}
}
return rtn;
})()
