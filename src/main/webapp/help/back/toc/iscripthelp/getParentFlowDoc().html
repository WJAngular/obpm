//跨表单主子流程，子流程表单某控件值为父流程文档Id(值脚本)
(function(){
var pfdocid = getItemValueAsString("获取ID");
if (pfdocid==null || pfdocid.trim().length()<=0){
   var pfdoc = getParentFlowDoc();
   if(pfdoc!=null){
      pfdocid = pfdoc.getId();
}
}
return pfdocid;
})()
